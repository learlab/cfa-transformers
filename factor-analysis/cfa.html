df <- read_csv(data_path, show_col_types = FALSE) %>% 
  rename_all(make.names)
library(lavaan)
library(lavaanPlot)

spec.grammar <- 
' human =~ Grammar_1 + Grammar_2
  llm =~ bert + deberta + xlm '

fit.grammar <- cfa(spec.grammar, data = df, std.lv=TRUE)
